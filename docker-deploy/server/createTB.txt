CREATE TABLE IF NOT EXISTS SYMBOL(
SYMBOL_NAME VARCHAR(256),
PRIMARY KEY (SYMBOL_NAME)
);

CREATE TABLE IF NOT EXISTS ACCOUNT(
ACCOUNT_ID BIGINT,
BALANCE DOUBLE PRECISION,
VERSION integer NOT NULL DEFAULT 1,
PRIMARY KEY (ACCOUNT_ID)
);

CREATE TABLE IF NOT EXISTS POSITION(
SYMBOL_NAME VARCHAR(256),
ACCOUNT_ID BIGINT,
AMOUNT DOUBLE PRECISION,
VERSION integer NOT NULL DEFAULT 1,
CONSTRAINT POSITION_PK PRIMARY KEY (SYMBOL_NAME, ACCOUNT_ID),
FOREIGN KEY (ACCOUNT_ID) REFERENCES ACCOUNT (ACCOUNT_ID) ON DELETE SET NULL ON UPDATE CASCADE,
FOREIGN KEY (SYMBOL_NAME) REFERENCES SYMBOL (SYMBOL_NAME) ON DELETE SET NULL ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS OPENORDER(
TRANS_ID BIGSERIAL,
SYMBOL_NAME VARCHAR(256),
ACCOUNT_ID BIGINT,
AMOUNT DOUBLE PRECISION,
LIMIT_PRICE DOUBLE PRECISION,
TIME BIGINT,
VERSION integer NOT NULL DEFAULT 1,
PRIMARY KEY (TRANS_ID),
FOREIGN KEY (ACCOUNT_ID) REFERENCES ACCOUNT (ACCOUNT_ID) ON DELETE SET NULL ON UPDATE CASCADE,
FOREIGN KEY (SYMBOL_NAME) REFERENCES SYMBOL (SYMBOL_NAME) ON DELETE SET NULL ON UPDATE CASCADE
);

CREATE TABLE IF NOT EXISTS EXECUTEDORDER(
EXECUTED_ID BIGSERIAL,
TRANS_ID BIGINT,
ACCOUNT_ID BIGINT,
AMOUNT DOUBLE PRECISION,
DEAL_PRICE DOUBLE PRECISION,
TIME BIGINT,
PRIMARY KEY (EXECUTED_ID)
);

CREATE TABLE IF NOT EXISTS CANCELEDORDER(
TRANS_ID BIGINT,
ACCOUNT_ID BIGINT,
AMOUNT DOUBLE PRECISION,
TIME BIGINT,
PRIMARY KEY (TRANS_ID)
);
